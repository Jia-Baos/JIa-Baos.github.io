#include <iostream>

class A
{
public:
	virtual void vfunc1() { puts("A::vfunc1"); }
	virtual void vfunc2() { puts("A::vfunc2"); }
	void func1() { puts("A::func1"); }
	void func2() { puts("A::func2"); }

private:
	int m_data1;
	int m_data2;
};

class B :public A
{
public:
	virtual void vfunc1() { puts("B::vfunc1"); }
	void func1() { puts("B::func1"); }
private:
	int m_data3;
};

class C :public B
{
public:
	virtual void vfunc2() { puts("C::vfunc2"); }
	void func2() { puts("C::func2"); }
private:
	int m_data1;
	int m_data4;
};

int main(int argc, char* argv[])
{
	std::cout << _MSVC_LANG << std::endl;
	A a;

	B b;
	C c;

	puts("-----------------");
	a.vfunc1();	// A::vfunc1
	a.vfunc2();	// A::vfunc2
	a.func1();	// A::func1
	a.func2();	// A::func2

	puts("-----------------");
	b.vfunc1();	// B::vfunc1
	b.vfunc2();	// A::vfunc2
	b.func1();	// B::func1
	b.func2();	// A::func2

	puts("-----------------");
	c.vfunc1();	// B::vfunc1
	c.vfunc2();	// C::vfunc2
	c.func1();	// B::func1
	c.func2();	// C::func2

	return 0;
}

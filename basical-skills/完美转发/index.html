完美转发

PUBLISHED ON JUL 12, 2023 / 3 MIN READ — ARTICLES

std::forward被称为完美转发，它的作用是保持原来的值属性不变。

#include <bits/stdc++.h>

template<typename T>
void print(T & t){
    std::cout << "left value" << std::endl;
}

template<typename T>
void print(T && t){
    std::cout << "right value" << std::endl;
}

template<typename T>
void testForward(T && v){
    print(v);
    print(std::forward<T>(v));
    print(std::move(v));
}

int main() {
    std::cout << __cplusplus << std::endl;

    testForward(1);
    /*
    * left value
    * right value
    * right value
    */

    int x = 1;
    testForward(x);
    /*
    * left value
    * left value
    * right value
    */

    return 0;
}

#include <bits/stdc++.h>

// 要求左值表达式
void print(int &t) {
    std::cout << "left value" << std::endl;
}

// 要求右值表达式，亡值、纯右值
void print(int &&t) {
    std::cout << "right value" << std::endl;
}

template<typename _Tp>
constexpr _Tp &&
forward(typename std::remove_reference<_Tp>::type &__t) noexcept {
    std::cout << "left value forward" << std::endl;
    return static_cast<_Tp &&>(__t);
}


template<typename _Tp>
constexpr _Tp &&
forward(typename std::remove_reference<_Tp>::type &&__t) noexcept {
    std::cout << "right value forward" << std::endl;
    static_assert(!std::is_lvalue_reference<_Tp>::value, "template argument"
                                                         " substituting _Tp must not be an lvalue reference type");
    return static_cast<_Tp &&>(__t);
}


int main() {
    std::cout << __cplusplus << std::endl;

    int val{};

    std::cout << "---------------" << std::endl;
    print(forward<int>(val));
    print(forward<int &>(val));
    print(forward<int &&>(val));

    // left value forward
    // right value
    // left value forward
    // left value
    // left value forward
    // right value

    std::cout << "---------------" << std::endl;
    print(forward<int>(0));
    //  print(forward<int &>(0));   // 静态检查报错
    print(forward<int &&>(0));

    //  right value forward
    //  right value
    //  right value forward
    //  right value

    return 0;
}

下面为 std::forward 和 std::move 的源码。

  /**
   *  @brief  Forward an lvalue.
   *  @return The parameter cast to the specified type.
   *
   *  This function is used to implement "perfect forwarding".
   */
  template<typename _Tp>
    _GLIBCXX_NODISCARD
    constexpr _Tp&&
    forward(typename std::remove_reference<_Tp>::type& __t) noexcept
    { return static_cast<_Tp&&>(__t); }

  /**
   *  @brief  Forward an rvalue.
   *  @return The parameter cast to the specified type.
   *
   *  This function is used to implement "perfect forwarding".
   */
  template<typename _Tp>
    _GLIBCXX_NODISCARD
    constexpr _Tp&&
    forward(typename std::remove_reference<_Tp>::type&& __t) noexcept
    {
      static_assert(!std::is_lvalue_reference<_Tp>::value, "template argument"
		    " substituting _Tp must not be an lvalue reference type");
      return static_cast<_Tp&&>(__t);
    }

  /**
   *  @brief  Convert a value to an rvalue.
   *  @param  __t  A thing of arbitrary type.
   *  @return The parameter cast to an rvalue-reference to allow moving it.
  */
  template<typename _Tp>
    _GLIBCXX_NODISCARD
    constexpr typename std::remove_reference<_Tp>::type&&
    move(_Tp&& __t) noexcept
    { return static_cast<typename std::remove_reference<_Tp>::type&&>(__t); }

分析。

  1. std::remove_reference

若类型 T 为引用类型，则提供成员 typedef type，其为 T 所引用的类型。否则 type 为 T。

template< class T > struct remove_reference      {typedef T type;};
template< class T > struct remove_reference<T&>  {typedef T type;};
template< class T > struct remove_reference<T&&> {typedef T type;};
  1. std::move

采用 std::remove_reference 去除引用后，再采用 static_cast 将 _t 强制转化为 右值引用类型。

  1. std::forward

采用 static_cast 进行强制转化的过程中触发引用折叠，具体表现为：

A& &	-> A&
A& &&	-> A&
A&& &	-> A&
A&& &&	-> A&&

简单来讲就是左值引用会传染，沾上一个左值引用就变成左值引用了。

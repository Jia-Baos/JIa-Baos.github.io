动态、静态类型

PUBLISHED ON JUL 15, 2023 / 2 MIN READ — ARTICLES

如果表达式的类型是类类型且至少包含有一个虚函数，则typeid操作符返回表达式的动态类型，需要在运行时计算；否则，typeid操作符返回表达式的静态类型，在编译时就可以计算。

#include <iostream>

struct Base {
	virtual ~Base() {}
};

struct Derived : Base {
};

int main(int argc, char* argv[]) {
	Derived derived;
	Base* ptr = &derived;

	puts("----------------");

	// (*ptr)的静态类型是B
	// (*ptr)的静态类型是D
	std::cout << std::boolalpha
		<< std::is_same_v<std::remove_reference_t<decltype(*ptr)>, Base> << std::endl;
	std::cout << std::boolalpha
		<< (typeid(*ptr) == typeid(Base)) << std::endl;

	// ----------------
	// true
	// false
}

#include <iostream>
#include <typeinfo>

#define OUTPUT(f)    std::cout << #f << "\t: " << typeid(f).name() << std::endl;

class BaseA {
};

class DeriveA : public BaseA {
};

class BaseB {
	virtual void f() {}
};

class DeriveB : public BaseB {
};

int main(int argc, char* argv[]) {
	std::cout << "-------直接处理类名-------" << std::endl;

	OUTPUT(BaseA);
	OUTPUT(DeriveA);
	OUTPUT(BaseB);
	OUTPUT(DeriveB);

	std::cout << "-------基类不含虚函数-------" << std::endl;

	BaseA baseA;
	DeriveA deriveA;
	OUTPUT(baseA);
	OUTPUT(deriveA);

	BaseA* pa;
	pa = &baseA;
	OUTPUT(*pa);
	OUTPUT(pa);
	pa = &deriveA;
	OUTPUT(*pa);
	OUTPUT(pa);

	std::cout << "-------基类含有虚函数-------" << std::endl;

	BaseB baseB;
	DeriveB deriveB;
	OUTPUT(baseB);
	OUTPUT(deriveB);

	BaseB* pb;
	pb = &baseB;
	OUTPUT(*pb);
	OUTPUT(pb);
	pb = &deriveB;
	OUTPUT(*pb);
	OUTPUT(pb);
}

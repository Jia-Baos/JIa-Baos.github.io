C++20管道运算符

PUBLISHED ON JUL 23, 2023 / 1 MIN READ — ARTICLES 
#include <iostream>
#include <ranges>


int main()
{
    auto const ints = { 0,1,2,3,4,5 };
    auto even = [](int i) { return 0 == i % 2; };
    auto square = [](int i) { return i * i; };

    // 组合视图的“管道”语法：
    for (int i : ints | std::views::filter(even) | std::views::transform(square)) {
        std::cout << i << ' ';
    }

    std::cout << std::endl;

    // 传统的“函数式”组合语法,filter是筛选，然后再用transform筛选
    for (int i : std::views::transform(std::views::filter(ints, even), square)) {
        std::cout << i << ' ';
    }
    std::cout << std::endl;

    auto f = std::views::transform([](int n) { return n *= n; });
    auto ret = ints | f;//并没有改变ints，只是返回一个序列
    for (auto i : ret) {
        std::cout << i << ' ';
    }
}

运算符重载实现管道运算符

#include <iostream>
#include <vector>
#include <algorithm>
#include <functional>

template<typename T, typename F>
	requires std::regular_invocable<F, T&>
std::vector<T>& operator|(std::vector<T>& v, F f)
{
	for (auto& iter : v)
	{
		f(iter);
	}
	return v;
}

int main(int argc, char* argv[])
{
	std::vector v{1, 2, 3, 4, 5};

	auto f1 = [](int& i) {i += i; };
	auto f2 = [](int& i) {i *= i; };
	std::function f3{[](const int& i) {std::cout << i << " "; } };

	v | f1 | f2 | f3;
}

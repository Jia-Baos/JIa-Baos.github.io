#include <bits/stdc++.h>

struct X {
    X() { puts("int"); }

    X(const int &) { puts("const int&"); }

    X(int &&) { puts("int&&"); }
};

X f() {
    return X();
}

int main(int argc, char *argv[]) {
    X x1 = 1;   // int&&
    puts("--------------");
    X x2 = f(); // int
}
